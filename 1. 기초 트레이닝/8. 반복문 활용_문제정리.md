def solution(picture, k):
    answer = []

    for i in range(len(picture)):
        temp = list(picture[i]) 
        for j in range(len(temp)):
            temp[j] *= k
        temp = ''.join(temp)    
        for _ in range(k):
            answer.append(temp)
    
    return answer

if __name__ == "__main__":
    picture = input()
    k = int(input())
    
    result = solution(picture, k)
    print(result)
